Create app-probes-service.yaml # Create a service for probe deployment
apiVersion: v1
kind: Service
metadata:
  name: app-probes
spec:
  selector:
    app: app-probes
  ports:
    - protocol: TCP
      port: 80
      targetPort: 80
  type: ClusterIP
